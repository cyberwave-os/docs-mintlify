        onSubmit={(e) => {
          e.preventDefault();
          const form = e.target;
          const formData = new FormData(form);
          const fullName = formData.get('fullName');
          const nameParts = fullName.split(' ');
          const submitButton = form.querySelector('button[type="submit"]');
          const statusDiv = document.getElementById('form-status');
          
          submitButton.disabled = true;
          submitButton.textContent = 'Joining...';
          statusDiv.innerHTML = '';
          
          fetch('https://api.cyberwave.com/api/v1/contact/', {
            method: 'POST',
            headers: {
              'Content-Type': 'application/json',
            },
            body: JSON.stringify({
              first_name: nameParts[0] || fullName,
              last_name: nameParts.slice(1).join(' ') || '',
              email: formData.get('email'),
              subject: 'Waitlist Registration',
              message: `Waitlist Registration:\n\n${formData.get('message') || 'No additional message provided.'}\n\nNote: This user has joined the waitlist for early access to new features.`

          .then(res => res.json())
          .then(data => {
            if (data.success) {
              statusDiv.innerHTML = '<p style="color: #00b5dd; font-weight: bold; margin: 0; padding: 0.75rem; background: #e6f7fb; border-radius: 0.5rem;">✓ Welcome to the waitlist! We\'ll be in touch soon.</p>';
              form.reset();
              setTimeout(() => {
                document.getElementById('waitlist-modal').style.display = 'none';
                statusDiv.innerHTML = '';
              }, 2000);
            } else {
              statusDiv.innerHTML = '<p style="color: #dc2626; margin: 0; padding: 0.75rem; background: #fee; border-radius: 0.5rem;">⚠ ' + (data.message || 'Something went wrong. Please try again.') + '</p>';
            }
          })
          .catch(err => {
            statusDiv.innerHTML = '<p style="color: #dc2626; margin: 0; padding: 0.75rem; background: #fee; border-radius: 0.5rem;">⚠ Failed to join waitlist. Please try again.</p>';
          })
          .finally(() => {
            submitButton.disabled = false;
            submitButton.textContent = 'Join Private Beta';
          });
